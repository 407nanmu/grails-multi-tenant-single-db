001    package grails.plugin.multitenant.core.ast;
002    
003    import grails.plugin.multitenant.singledb.hibernate.TenantFilterCfg;
004    
005    import java.lang.reflect.Modifier;
006    import java.util.ArrayList;
007    import java.util.List;
008    
009    import org.apache.commons.logging.Log;
010    import org.apache.commons.logging.LogFactory;
011    import org.codehaus.groovy.ast.ASTNode;
012    import org.codehaus.groovy.ast.ClassNode;
013    import org.codehaus.groovy.ast.expr.ConstantExpression;
014    import org.codehaus.groovy.ast.stmt.Statement;
015    import org.codehaus.groovy.control.CompilePhase;
016    import org.codehaus.groovy.control.SourceUnit;
017    import org.codehaus.groovy.grails.compiler.injection.GrailsASTUtils;
018    import org.codehaus.groovy.transform.ASTTransformation;
019    import org.codehaus.groovy.transform.GroovyASTTransformation;
020    
021    /**
022     * Adds a tenantId property to domain classes annotated MultiTenant.
023     */
024    @GroovyASTTransformation(phase = CompilePhase.CANONICALIZATION)
025    public class MultiTenantAST implements ASTTransformation {
026        
027        private static Log log = LogFactory.getLog(MultiTenantAST.class);
028            
029            @Override
030        public void visit(ASTNode[] astNodes, SourceUnit sourceUnit) {
031            for (ClassNode classNode : getClassAstNodes(astNodes)) {
032                    log.debug("Processing " + classNode.getName());
033                    if (isMissingTenantId(classNode)) {
034                            addTenantProperty(classNode);
035                    }
036            }
037        }
038        
039        private List<ClassNode> getClassAstNodes(ASTNode[] astNodes) {
040            List<ClassNode> classAstNodes = new ArrayList<ClassNode>();
041            for (ASTNode astNode : astNodes) {
042                if (astNode instanceof ClassNode) {
043                    ClassNode classAstNode = (ClassNode) astNode;
044                    classAstNodes.add(classAstNode);
045                }
046            }
047            
048            return classAstNodes;
049        }
050        
051        private boolean isMissingTenantId(ClassNode node) {
052            return !GrailsASTUtils.hasOrInheritsProperty(node, "tenantId");
053        }
054        
055        private void addTenantProperty(ClassNode node) {
056            log.debug("Adding tenant id property to " + node.getName());
057            ClassNode integerType = new ClassNode(Integer.class);
058            ConstantExpression defaultValue = new ConstantExpression(0);
059            Statement getterBlock = null;
060            Statement setterBlock = null;
061            node.addProperty(TenantFilterCfg.TENANT_ID_FIELD_NAME, Modifier.PUBLIC, integerType, 
062                            defaultValue, getterBlock, setterBlock);
063        }
064        
065    }
