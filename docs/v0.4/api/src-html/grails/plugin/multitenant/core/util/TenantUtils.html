001    package grails.plugin.multitenant.core.util;
002    
003    import grails.plugin.multitenant.core.CurrentTenant;
004    import groovy.lang.Closure;
005    
006    import org.hibernate.SessionFactory;
007    import org.hibernate.classic.Session;
008    
009    /**
010     * @deprecated
011     * @author Kim A. Betti <kim@developer-b.com>
012     */
013    @Deprecated
014    public class TenantUtils {
015    
016        private static CurrentTenant currentTenant;
017        private static SessionFactory sessionFactory;
018    
019        public void withTenantId(Integer temporaryTenantId, Closure closure) {
020            Integer previousTenantId = currentTenant.get();
021            Session currentSession = sessionFactory.getCurrentSession();
022            currentSession.flush();
023    
024            try {
025                currentTenant.set(temporaryTenantId);
026                closure.call();
027                currentSession.flush(); // Force events to happen with the expected id
028            } finally {
029                currentTenant.set(previousTenantId);
030            }
031        }
032    
033        public void setCurrentTenant(CurrentTenant currentTenant) {
034            TenantUtils.currentTenant = currentTenant;
035        }
036    
037        public void setSessionFactory(SessionFactory sessionFactory) {
038            TenantUtils.sessionFactory = sessionFactory;
039        }
040    
041    }
