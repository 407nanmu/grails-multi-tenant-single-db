001    package grails.plugin.multitenant.core.spring;
002    
003    import grails.plugin.multitenant.core.CurrentTenant;
004    
005    import java.util.Map;
006    import java.util.concurrent.ConcurrentHashMap;
007    
008    import org.springframework.beans.factory.ObjectFactory;
009    import org.springframework.beans.factory.config.Scope;
010    import org.springframework.context.ApplicationContext;
011    
012    /**
013     * 
014     * @author Kim A. Betti
015     */
016    public class TenantScope implements Scope {
017    
018        public static final String NAME = "tenant";
019    
020        private CurrentTenant currentTenant;
021        private ApplicationContext applicationContext;
022    
023        private ConcurrentHashMap<Integer, Map<String, Object>> tenantBeanCache = new ConcurrentHashMap<Integer, Map<String, Object>>(100);
024    
025        /**
026         * Return the object with the given name from the underlying scope, creating
027         * it if not found in the underlying storage mechanism.
028         */
029        @Override
030        public Object get(String name, ObjectFactory<?> objectFactory) {
031            Integer tenantId = currentTenant.get();
032            Map<String, Object> beanCache = getBeanCacheForTenant(tenantId);
033            if (!beanCache.containsKey(name)) {
034                beanCache.put(name, applicationContext.getBean(name + "_prototype"));
035            }
036    
037            return beanCache.get(name);
038        }
039    
040        private Map<String, Object> getBeanCacheForTenant(Integer tenantId) {
041            if (!tenantBeanCache.containsKey(tenantId)) {
042                tenantBeanCache.put(tenantId, new ConcurrentHashMap<String, Object>(100));
043            }
044    
045            return tenantBeanCache.get(tenantId);
046        }
047    
048        /**
049         * Return the conversation ID for the current underlying scope, if any.
050         */
051        @Override
052        public String getConversationId() {
053            throw new RuntimeException("getConversationId is not implemented");
054        }
055    
056        /**
057         * Register a callback to be executed on destruction of the specified object
058         * in the scope (or at destruction of the entire scope, if the scope does
059         * not destroy individual objects but rather only terminates in its
060         * entirety).
061         */
062        @Override
063        public void registerDestructionCallback(String name, Runnable callback) {
064            throw new RuntimeException("registerDestructionCallback " + name + " is not implemented");
065        }
066    
067        /**
068         * Remove the object with the given name from the underlying scope.
069         */
070        @Override
071        public Object remove(String name) {
072            throw new RuntimeException("remove " + name + " is not implemented");
073        }
074    
075        /**
076         * Resolve the contextual object for the given key, if any.
077         */
078        @Override
079        public Object resolveContextualObject(String key) {
080            throw new RuntimeException("resolveContextualObject " + key + " is not implemented");
081        }
082    
083        public void setCurrentTenant(CurrentTenant currentTenant) {
084            this.currentTenant = currentTenant;
085        }
086    
087        public void setApplicationContext(ApplicationContext applicationContext) {
088            this.applicationContext = applicationContext;
089        }
090    
091    }
