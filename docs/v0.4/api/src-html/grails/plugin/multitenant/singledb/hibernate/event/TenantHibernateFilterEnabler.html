001    package grails.plugin.multitenant.singledb.hibernate.event;
002    
003    import grails.plugin.multitenant.core.CurrentTenant;
004    import grails.plugin.multitenant.singledb.hibernate.TenantFilterCfg;
005    import grails.plugins.hawkeventing.Event;
006    import grails.plugins.hawkeventing.annotation.Consuming;
007    import grails.plugins.hawkeventing.annotation.HawkEventConsumer;
008    
009    import org.apache.commons.logging.Log;
010    import org.apache.commons.logging.LogFactory;
011    import org.hibernate.SessionFactory;
012    import org.hibernate.classic.Session;
013    import org.springframework.transaction.support.TransactionSynchronizationManager;
014    
015    /**
016     * Subscribes itself to hibernate.sessionCreated events.
017     * Enables the tenant Hibernate filter with the current tenant id (if any)
018     * 
019     * @author Kim A. Betti
020     */
021    @HawkEventConsumer
022    public class TenantHibernateFilterEnabler {
023    
024        private static Log log = LogFactory.getLog(TenantHibernateFilterEnabler.class);
025    
026        private SessionFactory sessionFactory;
027        private CurrentTenant currentTenant;
028    
029        @Consuming("hibernate.sessionCreated")
030        public void newHibernateSessionCreated(Event event) {
031            Session newSession = (Session) event.getPayload();
032            Integer currentTenantId = currentTenant.get();
033            enableHibernateFilter(newSession, currentTenantId);
034        }
035    
036        @Consuming(CurrentTenant.TENANT_CHANGE_EVENT)
037        public void currentTenantUpdated(Event event) {
038            if (TransactionSynchronizationManager.hasResource(sessionFactory)) {
039                Integer updatedTenantId = (Integer) event.getPayload();
040                Session currentSession = sessionFactory.getCurrentSession();
041                enableHibernateFilter(currentSession, updatedTenantId);
042            }
043        }
044    
045        public void enableHibernateFilter(Session session, Integer tenantId) {
046            session.enableFilter(TenantFilterCfg.TENANT_FILTER_NAME).setParameter (
047                    TenantFilterCfg.TENANT_ID_PARAM_NAME, tenantId);
048        }
049    
050        public void setCurrentTenant(CurrentTenant currentTenant) {
051            this.currentTenant = currentTenant;
052        }
053    
054        public void setSessionFactory(SessionFactory sessionFactory) {
055            this.sessionFactory = sessionFactory;
056        }
057    
058    }
