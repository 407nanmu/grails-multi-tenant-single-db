001    package grails.plugin.multitenant.singledb.hibernate;
002    
003    import grails.plugin.hibernatehijacker.hibernate.HibernateConfigPostProcessor;
004    import grails.plugin.hibernatehijacker.hibernate.events.HibernateEventUtil;
005    import grails.plugin.multitenant.core.MultiTenantContext;
006    import grails.plugin.multitenant.core.hibernate.event.TenantHibernateEventListener;
007    
008    import java.util.ArrayList;
009    import java.util.List;
010    import java.util.Map;
011    
012    import org.apache.commons.logging.Log;
013    import org.apache.commons.logging.LogFactory;
014    import org.codehaus.groovy.grails.commons.GrailsDomainClass;
015    import org.codehaus.groovy.grails.validation.ConstrainedProperty;
016    import org.hibernate.HibernateException;
017    import org.hibernate.cfg.Configuration;
018    import org.hibernate.engine.FilterDefinition;
019    import org.hibernate.event.EventListeners;
020    import org.hibernate.mapping.PersistentClass;
021    
022    /**
023     * Defines the Hibernate filter.
024     * 
025     * The Hibernate Hijacker detects beans implementing
026     * HibernateConfigPostProcessor and will make sure that the Configuration
027     * instances is passed through this instance when the application is starting.
028     * 
029     * @author Kim A. Betti <kim@developer-b.com>
030     */
031    public class TenantHibernateFilterConfigurator implements HibernateConfigPostProcessor {
032    
033        private static Log log = LogFactory.getLog(TenantHibernateFilterConfigurator.class);
034    
035        // TODO: Move this somewhere else..? (also used in AST).
036        public final static String TENANT_ID_FIELD_NAME = "tenantId";
037    
038        private FilterDefinition multiTenantHibernateFilter;
039        private TenantHibernateEventListener tenantHibernateEventListener;
040        private MultiTenantContext multiTenantContext;
041    
042        @Override
043        public void doPostProcessing(Configuration configuration) throws HibernateException {
044            log.debug("Configuring multi-tenant Hibernate filter");
045    
046            createFilterDefinition(configuration);
047            enrichMultiTenantDomainClasses(configuration);
048            activateTenantEventListener(configuration);
049        }
050    
051        private void createFilterDefinition(Configuration configuration) {
052            log.debug("Defining Multi Tenant Hibernate filer");
053            configuration.addFilterDefinition(multiTenantHibernateFilter);
054        }
055    
056        private void enrichMultiTenantDomainClasses(Configuration configuration) {
057            List<String> classNames = new ArrayList<String>();
058            List<GrailsDomainClass> multiTenantDomainClasses = multiTenantContext.getMultiTenantDomainClasses();
059            for (GrailsDomainClass domainClass : multiTenantDomainClasses) {
060                classNames.add(domainClass.getClass().getSimpleName());
061                addDomainFilter(domainClass, configuration);
062                addTenantIdConstraints(domainClass);
063            }
064    
065            log.debug("Added multi tenant functionality to: " + classNames);
066        }
067    
068        private void addDomainFilter(GrailsDomainClass domainClass, Configuration configuration) {
069            PersistentClass entity = configuration.getClassMapping(domainClass.getFullName());
070            String filterName = multiTenantHibernateFilter.getFilterName();
071            String condition = multiTenantHibernateFilter.getDefaultFilterCondition();
072            entity.addFilter(filterName, condition);
073        }
074    
075        @SuppressWarnings("unchecked")
076        private void addTenantIdConstraints(GrailsDomainClass domainClass) {
077            Map<String, ConstrainedProperty> constraints = domainClass.getConstrainedProperties();
078            constraints.get(TENANT_ID_FIELD_NAME).applyConstraint(ConstrainedProperty.NULLABLE_CONSTRAINT, false);
079        }
080    
081        private void activateTenantEventListener(Configuration configuration) {
082            EventListeners eventListeners = configuration.getEventListeners();
083            HibernateEventUtil.addListener(eventListeners, tenantHibernateEventListener);
084        }
085    
086        public void setMultiTenantHibernateFilter(FilterDefinition multiTenantHibernateFilter) {
087            this.multiTenantHibernateFilter = multiTenantHibernateFilter;
088        }
089    
090        public void setTenantHibernateEventListener(TenantHibernateEventListener tenantHibernateEventListener) {
091            this.tenantHibernateEventListener = tenantHibernateEventListener;
092        }
093    
094        public void setMultiTenantContext(MultiTenantContext multiTenantContext) {
095            this.multiTenantContext = multiTenantContext;
096        }
097    
098    }
