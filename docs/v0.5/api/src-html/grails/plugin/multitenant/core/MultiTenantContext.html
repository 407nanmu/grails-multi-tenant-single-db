001    package grails.plugin.multitenant.core;
002    
003    import grails.plugin.multitenant.core.annotation.MultiTenant;
004    import grails.plugin.multitenant.core.exception.TenantException;
005    
006    import java.lang.annotation.Annotation;
007    import java.util.ArrayList;
008    import java.util.List;
009    
010    import org.codehaus.groovy.grails.commons.GrailsApplication;
011    import org.codehaus.groovy.grails.commons.GrailsClass;
012    import org.codehaus.groovy.grails.commons.GrailsDomainClass;
013    import org.springframework.beans.factory.InitializingBean;
014    
015    /**
016     * This class was created to get rid of TenantUtils and make some
017     * other classes independent of grailsApplication.
018     * 
019     * @author Kim A. Betti
020     */
021    public class MultiTenantContext implements InitializingBean {
022    
023        private GrailsApplication grailsApplication;
024    
025        private List<GrailsDomainClass> multiTenantDomainClasses;
026    
027        private Class<? extends Tenant> tenantClass;
028    
029        @Override
030        public void afterPropertiesSet() {
031            this.multiTenantDomainClasses = findMultiTenantDomainClasses();
032            this.tenantClass = findTenantClass();
033        }
034    
035        public List<GrailsDomainClass> findMultiTenantDomainClasses() {
036            List<GrailsDomainClass> domainClasses = new ArrayList<GrailsDomainClass>();
037            for (GrailsClass grailsClass : grailsApplication.getArtefacts("Domain")) {
038                Class<?> realClass = grailsClass.getClazz();
039                if (hasMultiTenantAnnotation(realClass)) {
040                    domainClasses.add((GrailsDomainClass)grailsClass);
041                }
042            }
043    
044            return domainClasses;
045        }
046    
047        public boolean hasMultiTenantAnnotation(Class<?> clazz) {
048            Annotation[] annotations = clazz.getAnnotations();
049            for (Annotation annotation : annotations) {
050                if (annotation instanceof MultiTenant) {
051                    return true;
052                }
053            }
054    
055            return false;
056        }
057    
058        @SuppressWarnings("unchecked")
059        public Class<? extends Tenant> findTenantClass() {
060            Class<? extends Tenant> tenantDomainClass = null;
061            for (GrailsClass grailsClass : grailsApplication.getArtefacts("Domain")) {
062                Class<?> realDomainClass = grailsClass.getClazz();
063                if (Tenant.class.isAssignableFrom(realDomainClass)) {
064                    if (tenantDomainClass != null) {
065                        throw new TenantException(String.format("Found multiple implementations of the Tenant interface (%s and %s)",
066                                tenantDomainClass.getName(), realDomainClass.getName()));
067                    }
068    
069                    tenantDomainClass = (Class<? extends Tenant>) realDomainClass;
070                }
071            }
072    
073            return tenantDomainClass;
074        }
075    
076        public List<GrailsDomainClass> getMultiTenantDomainClasses() {
077            return multiTenantDomainClasses;
078        }
079    
080        public Class<? extends Tenant> getTenantClass() {
081            return this.tenantClass;
082        }
083    
084        public boolean isMultiTenantDomainClass(Class<?> clazz) {
085            for (GrailsDomainClass domainClass : multiTenantDomainClasses) {
086                if (domainClass.getClazz() == clazz) {
087                    return true;
088                }
089            }
090    
091            return false;
092        }
093    
094        public void setGrailsApplication(GrailsApplication grailsApplication) {
095            this.grailsApplication = grailsApplication;
096        }
097    
098    }
