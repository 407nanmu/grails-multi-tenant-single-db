001    package grails.plugin.multitenant.singledb.hibernate;
002    
003    import grails.plugin.multitenant.core.CurrentTenant;
004    import grails.plugin.multitenant.core.MultiTenantDomainClass;
005    import grails.plugin.multitenant.core.ast.MultiTenantAST;
006    import grails.plugin.multitenant.core.exception.NoCurrentTenantException;
007    import grails.plugin.multitenant.core.exception.TenantException;
008    import grails.plugin.multitenant.core.exception.TenantSecurityException;
009    
010    import java.util.HashMap;
011    import java.util.Map;
012    
013    import org.hibernate.HibernateException;
014    import org.hibernate.event.LoadEvent;
015    import org.hibernate.event.LoadEventListener;
016    import org.hibernate.event.PreInsertEvent;
017    import org.hibernate.event.PreInsertEventListener;
018    import org.hibernate.event.PreUpdateEvent;
019    import org.hibernate.event.PreUpdateEventListener;
020    import org.hibernate.tuple.StandardProperty;
021    import org.hibernate.tuple.entity.EntityMetamodel;
022    import org.slf4j.Logger;
023    import org.slf4j.LoggerFactory;
024    
025    /**
026     * Listens for pre-insert, pre-update and load / fetch Hibernate events.
027     * If the domain class related to the event is a multi-tenant class we apply
028     * the relevant constraints.
029     * @author Kim A. Betti
030     */
031    @SuppressWarnings("serial")
032    public class TenantHibernateEventListener implements PreInsertEventListener, PreUpdateEventListener, LoadEventListener {
033    
034        private static Logger log = LoggerFactory.getLogger(TenantHibernateFilterConfigurator.class);
035    
036        private CurrentTenant currentTenant;
037    
038        // The PostLoad event only contains the class name so reflection is used to
039        // load the corresponding class. This is obviously expensive so we cache the result.
040        private Map<String, Class<?>> reflectedCache = new HashMap<String, Class<?>>();
041    
042        // We need to get the index of the tenantId property.
043        // This is another expensive operation so the result is cached here.
044        private Map<Class<?>, Integer> entityParamIndexCache = new HashMap<Class<?>, Integer>();
045    
046        /**
047         * One important thing to know. It's not enough to update the
048         * entity instance with the new tenant-id. Hibernate will _not_ pick up
049         * on this and will therefore not save the updated tenant id to database.
050         * 
051         * We have to get hold of the JPA meta model, find the index of the
052         * tenantId field and update the entity state in the event.
053         */
054        @Override
055        public boolean onPreInsert(PreInsertEvent event) {
056            if (isMultiTenantEntity(event.getEntity())) {
057                Integer currentTenantId = currentTenant.get();
058                if (currentTenantId == null) {
059                    throw new NoCurrentTenantException("Tried to save multi-tenant domain class '"
060                            + event.getEntity().getClass().getSimpleName() + "', but no tenant is set");
061                }
062    
063                MultiTenantDomainClass entity = (MultiTenantDomainClass) event.getEntity();
064                entity.setTenantId(currentTenantId);
065                updateTenantIdOnEvent(event, currentTenantId);
066            }
067    
068            return false;
069        }
070    
071        private void updateTenantIdOnEvent(PreInsertEvent event, Integer currentTenantId) {
072            int paramIndex = getTenantIdParamIndex(event);
073            event.getState()[paramIndex] = currentTenantId;
074        }
075    
076        private int getTenantIdParamIndex(PreInsertEvent event) {
077            Class<?> entityClass = event.getEntity().getClass();
078            if (!entityParamIndexCache.containsKey(entityClass)) {
079                EntityMetamodel metaModel = event.getPersister().getEntityMetamodel();
080                int propertyIndex = getPropertyIndex(metaModel, MultiTenantAST.TENANT_ID_FIELD_NAME);
081                entityParamIndexCache.put(entityClass, propertyIndex);
082            }
083    
084            return entityParamIndexCache.get(entityClass);
085        }
086    
087        private int getPropertyIndex(EntityMetamodel metaModel, String propertyName) {
088            int i = 0;
089            StandardProperty[] properties = metaModel.getProperties();
090            for (StandardProperty property : properties) {
091                if (property.getName().equals(propertyName)) {
092                    return i;
093                }
094    
095                i++;
096            }
097    
098            throw new TenantException("Unable to find property index for: " + propertyName);
099        }
100    
101        @Override
102        public boolean onPreUpdate(PreUpdateEvent event) {
103            if (isMultiTenantEntity(event.getEntity())) {
104                Integer currentTenantId = currentTenant.get();
105                MultiTenantDomainClass entity = (MultiTenantDomainClass) event.getEntity();
106                Integer entityTenantId = entity.getTenantId();
107                if (!currentTenantId.equals(entityTenantId)) {
108                    throw new TenantSecurityException("Tried to update '" + event.getEntity() + "' with another tenant id. Expected "
109                            + currentTenantId + ", found " + entityTenantId, currentTenantId, entityTenantId);
110                }
111            }
112    
113            return false;
114        }
115    
116        @Override
117        public void onLoad(LoadEvent event, LoadType type) throws HibernateException {
118            Class<?> entityClass = getClassFromName(event.getEntityClassName());
119            Object result = event.getResult();
120            if (result != null && isMultiTenantEntity(result)) {
121                Integer currentTenantId = currentTenant.get();
122                Integer loadedTenantId = ((MultiTenantDomainClass) result).getTenantId();
123                if (!currentTenantId.equals(loadedTenantId) && !event.isAssociationFetch()) {
124                    log.warn("Tried to load entity '" + entityClass.getSimpleName() + "' from other tenant, expected " + currentTenantId + ", found " + loadedTenantId);
125                    event.setResult(null);
126                }
127            }
128        }
129    
130        private Class<?> getClassFromName(String className) {
131            if (!reflectedCache.containsKey(className)) {
132                try {
133                    Class<?> aClass = getClass().getClassLoader().loadClass(className);
134                    reflectedCache.put(className, aClass);
135                } catch (ClassNotFoundException ex) {
136                    String message = "Could not find class " + className;
137                    throw new TenantException(message, ex);
138                }
139            }
140    
141            return reflectedCache.get(className);
142        }
143    
144        private boolean isMultiTenantEntity(Object entity) {
145            return entity instanceof MultiTenantDomainClass;
146        }
147    
148        public void setCurrentTenant(CurrentTenant currentTenant) {
149            this.currentTenant = currentTenant;
150        }
151    
152    }
